require('dotenv').config();
const path = require('path');
const fs = require('fs'); // Moved up top for cleanliness
const express = require('express');
const session = require('express-session');

const app = express();
const PORT = process.env.PORT || 3000; // Added a fallback just in case

// 1. MIDDLEWARE
app.use(express.json());
app.use(session({
  secret: process.env.SESSION_SECRET || 'fallback-secret', 
  resave: false,
  saveUninitialized: false
}));

// Serve static files
app.use(express.static(path.join(__dirname, 'public')));

// Helper path to your users file
const usersFilePath = path.join(__dirname, 'data', 'users.json');

// --- 2. AUTHENTICATION ROUTES ---

// GET: Check session status (Already in your code)
app.get('/api/me', (req, res) => {
  if (req.session.user) {
    res.status(200).json(req.session.user);
  } else {
    res.status(401).send("Unauthorized");
  }
});

// POST: Register new user (Already in your code)
app.post('/register', (req, res) => {
  const { username, password } = req.body;
  let users = [];
  try {
    const data = fs.readFileSync(usersFilePath, 'utf8');
    users = JSON.parse(data || '[]');
  } catch (err) {
    return res.status(500).send("Error reading database.");
  }

  if (users.find(u => u.username === username)) {
    return res.status(400).send("Username already taken.");
  }

  users.push({ username, password });
  fs.writeFileSync(usersFilePath, JSON.stringify(users, null, 2));
  res.sendStatus(200);
});

// --- STEP 3: THE MISSING LOGIN ROUTE ---
app.post('/login', (req, res) => {
  const { username, password } = req.body;
  
  let users = [];
  try {
    const data = fs.readFileSync(usersFilePath, 'utf8');
    users = JSON.parse(data || '[]');
  } catch (err) {
    return res.status(500).send("Error reading database.");
  }

  // Find user with matching name AND password
  const user = users.find(u => u.username === username && u.password === password);

  if (user) {
    // CRITICAL: Save the user to the session so the frontend checkAuth() works!
    req.session.user = { username: user.username };
    res.sendStatus(200);
  } else {
    res.status(401).send("Invalid username or password.");
  }
});

// --- 3. START SERVER ---
// Always put app.listen at the very bottom
app.listen(PORT, '0.0.0.0', () => {
  console.log(`Server running on port ${PORT}`);
});
